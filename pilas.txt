bool isEmpty(Nodo *pila){

AGREGA UN NODO A LA PILA

void push(Nodo **pila, int dato){
    Nodo *nodo = (Nodo*)malloc(sizeof(Nodo));
    nodo->dato = dato;
    nodo->siguiente = *pila;
    *pila = nodo;
}


REMUEVE UN NODO DE LA PILA(CUAL? NO SE, SIEMPRE ME PREGUNTO SI ES EL PRIMERO, EL ULTIMO)

int pop(Nodo **pila){
    int valor = (*pila)->dato;
    Nodo *aux = *pila;
    *pila = (*pila)->siguiente;
    free(aux);
    return valor;
}
